\unprotect
\def\doctitle#1{\gdef\@title{#1}}
\def\author#1{\gdef\@author{#1}}
\def\date#1{\gdef\@date{#1}}
\date{\currentdate}  % Default to today unless specified otherwise.

\def\maketitle{%
  \startalignment[center]
    \blank[force,2*big]
      {\switchtobodyfont[18pt] \@title\par}
    \blank[3*medium]
      {\switchtobodyfont[14pt] \@author\par}
    \blank[2*medium]
      {\switchtobodyfont[14pt] \@date\par}
    \blank[3*medium]
  \stopalignment}
\protect
